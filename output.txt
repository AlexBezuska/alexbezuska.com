========== START OF ./static-site-generator.js ==========
var fs = require("fs");
var path = require("path");
var handlebars = require("handlebars");

var site = require("./site.json");

var src = "./src";
var dest = "./dest";

makeDirIfNotExist(dest);

var template = fs.readFileSync(path.join(src, "index.hbs"), "utf8");
var compiledTemplate = handlebars.compile(template);
var html = compiledTemplate(site);

fs.writeFileSync(path.join(dest, "index.html"), html);

function makeDirIfNotExist(filePath) {
  if (!fs.existsSync(filePath)) {
    fs.mkdirSync(filePath, { recursive: true });
  }
}

  function scrolled(direction, selector){
    var scroll = $(window).scrollTop(); // how many pixels you've scrolled
    var os = $(selector).offset().top; // pixels to the top of div1
    var ht = $(selector).height(); // height of div1 in pixels
    if (direction === "below") return scroll > os + ht;
    if (direction === "above") return scroll < os + ht;
  }
